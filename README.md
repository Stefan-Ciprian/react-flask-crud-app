# react-flask-base

Requirements:
    
    $ python3.10, python3.10-dev, python3.10-venvn, node

## Setup
    $ sudo apt install python3.10-dev
    $ sudo apt install python3.10-venv
    $ python3.10 -m venv env
    $ source env/bin/activate
    $ pip install -r requirements.txt

## Running Locally
    $ yarn start-api
    $ yarn start
