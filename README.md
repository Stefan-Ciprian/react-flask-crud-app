# react-flask-base

Requirements:
    
    $ python3.10, python3.10-dev, python3.10-venvn, node.js v17

## Setup
    $ sudo apt install python3.10-dev
    $ sudo apt install python3.10-venv
    $ python3.10 -m venv env
    $ source env/bin/activate
    $ pip install -r requirements.txt

## Run tests
    $ ./run_tests.sh

## Running Locally
    $ ./run.sh
    $ yarn start
